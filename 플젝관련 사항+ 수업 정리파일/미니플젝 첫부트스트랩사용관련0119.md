좋아요 👍
아래는 React + Bootstrap(react-bootstrap) 으로
👉 **플랫폼용 대시보드 메인화면을 “최소 구조 + 실무 느낌”**으로 만드는 예제입니다.

📌 목표

KPI 카드

차트 영역

테이블(최근 데이터)

실무에서 바로 확장 가능

1️⃣ 사전 설치
npm install react-bootstrap bootstrap
npm install chart.js react-chartjs-2


main.jsx 또는 index.js

import 'bootstrap/dist/css/bootstrap.min.css';

2️⃣ 대시보드 기본 구조
📁 폴더 구조
src/
 ├─ pages/
 │   └─ Dashboard.jsx
 ├─ components/
 │   ├─ KpiCard.jsx
 │   ├─ EmissionChart.jsx
 │   └─ RecentTable.jsx

3️⃣ KPI 카드 컴포넌트
components/KpiCard.jsx
import { Card } from "react-bootstrap";

const KpiCard = ({ title, value, unit }) => {
  return (
    <Card className="shadow-sm">
      <Card.Body>
        <Card.Title className="text-muted fs-6">{title}</Card.Title>
        <h3 className="fw-bold">
          {value} <small className="fs-6">{unit}</small>
        </h3>
      </Card.Body>
    </Card>
  );
};

export default KpiCard;

4️⃣ 차트 컴포넌트
components/EmissionChart.jsx
import { Line } from "react-chartjs-2";
import {
  Chart as ChartJS,
  LineElement,
  CategoryScale,
  LinearScale,
  PointElement
} from "chart.js";

ChartJS.register(LineElement, CategoryScale, LinearScale, PointElement);

const EmissionChart = () => {
  const data = {
    labels: ["2019", "2020", "2021", "2022", "2023"],
    datasets: [
      {
        label: "탄소 배출량",
        data: [22, 20, 18, 15, 13],
        borderColor: "#0d6efd",
        tension: 0.3
      }
    ]
  };

  return <Line data={data} />;
};

export default EmissionChart;

6️⃣ 대시보드 메인 페이지
pages/Dashboard.jsx
import { Container, Row, Col, Card } from "react-bootstrap";
import KpiCard from "../components/KpiCard";
import EmissionChart from "../components/EmissionChart";
import RecentTable from "../components/RecentTable";

const Dashboard = () => {
  return (
    <Container fluid className="p-4">
      <h4 className="mb-4 fw-bold">대시보드</h4>

      {/* KPI */}
      <Row className="mb-4">
        <Col md={3}><KpiCard title="총 배출량" value="15,230" unit="tCO₂" /></Col>
        <Col md={3}><KpiCard title="목표 달성률" value="75" unit="%" /></Col>
        <Col md={3}><KpiCard title="올해 감축량" value="-1,200" unit="t" /></Col>
        <Col md={3}><KpiCard title="감축 목표" value="30" unit="%" /></Col>
      </Row>

      {/* 차트 + 테이블 */}
      <Row>
        <Col md={8}>
          <Card className="shadow-sm mb-4">
            <Card.Body>
              <Card.Title>연도별 탄소 배출 추이</Card.Title>
              <EmissionChart />
            </Card.Body>
          </Card>
        </Col>

        <Col md={4}>
          <Card className="shadow-sm">
            <Card.Body>
              <Card.Title>최근 데이터</Card.Title>
              <RecentTable />
            </Card.Body>
          </Card>
        </Col>
      </Row>
    </Container>
  );
};

export default Dashboard;
